\renewcommand{\name}{ \section*{Reimans sum} }
\renewcommand{\definition}{
	\paragraph{Definition}
	\begin{paragraph}
	\noindent
	The Reiman sum for a function $f$ on a closed interval $[a,b]$ is
	$$\sum_{k=1}^{n} f(c_k) \Delta_k $$ 
	where:
	The set of $\Delta$s is all subintervals of a partition $P$ of $[a,b]$
	and 
	$$ c_k \in \left[ \sum_{j=1}^{k-1} \Delta_k  ,\sum_{j=1}^{k} \Delta_k \right] $$
 	\end{paragraph}
 }
