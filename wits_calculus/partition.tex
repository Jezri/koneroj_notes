\renewcommand{\name}{Partition}
\renewcommand{\definition}{
	A parition for a closed interval $[a,b]$ is a set of point ($x_0 \text{ to }x_n$ ) such that:
	$$ a=x_0, \quad  x_k<x_{k+1} \quad \text{and} \quad b = x_n$$
}
