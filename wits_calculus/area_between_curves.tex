\renewcommand{\definition}{}
\renewcommand{\result}{
\paragraph{Result}
\noindent
\begin{paragraph}
The area between to curves $y = f(x)$ and $y=g(x)$ over a closed interval $[a,b]$is equal to 
$$\int_a^b f(x)-g(x) dx$$

\end{paragraph}
